賓客資料

try { console.log("正在初始化 Firebase..."); firebase.initializeApp(firebaseConfig); console.log("Firebase 初始化成功"); const db = firebase.database(); console.log("資料庫已定義"); // 從 URL 獲取 ID const urlParams = new URLSearchParams(window.location.search); const id = urlParams.get('id'); if (!id) { console.error("錯誤：無效的 ID"); document.getElementById('error-msg').innerText = "錯誤：無效的 ID"; return; } console.log("正在讀取資料，ID:", id); db.ref('records/' + id).once('value').then(snapshot => { const data = snapshot.val(); if (data) { console.log("資料讀取成功:", data); document.getElementById('data').innerHTML = `

纜車車箱號碼: ${data.carNumber}

賓客姓名: ${data.guestName}

健康狀況: ${data.healthStatus}

`; } else { console.error("錯誤：無此資料"); document.getElementById('error-msg').innerText = "錯誤：無此資料"; } }).catch(error => { console.error("資料讀取失敗:", error); document.getElementById('error-msg').innerText = "錯誤：資料讀取失敗: " + error.message; }); } catch (error) { console.error("Firebase 初始化失敗:", error); document.getElementById('error-msg').innerText = "錯誤：Firebase 初始化失敗: " + error.message; }
